CREATE DATABASE IF NOT EXISTS dbusiness DEFAULT CHARACTER SET utf8 ;

USE dbusiness;

CREATE TABLE dbusiness.CUSTOMER(
CUSTOMER_ID INT( 10 ) NOT NULL AUTO_INCREMENT ,
TITLE VARCHAR( 10 ) NOT NULL ,
NAME VARCHAR( 20 ) NOT NULL ,
PRIMARY KEY ( CUSTOMER_ID )
);

CREATE TABLE dbusiness.CUSTOMER_DETAIL(
CUSTOMER_ID INT( 10 ) NOT NULL AUTO_INCREMENT ,
ADDRESS VARCHAR( 50 ) DEFAULT NULL,
TEL VARCHAR( 20 ) DEFAULT NULL ,
CREATION_DATE DATE ,
PRIMARY KEY ( CUSTOMER_ID ),
CONSTRAINT FK_CUST FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMER (CUSTOMER_ID)
);
